


MEMBER [Measures].[DSRowNumber] AS
Rank(
	StrToTuple(
		"( " +
		Generate(
			Head(Axis(1), Axis(1).Item(0).Count ) AS RN,
			"Axis(1).Item(0).Item(" + CStr( RN.CurrentOrdinal - 1 ) + ").Hierarchy.CurrentMember",
			", "
		) +
		" )"
	),
	Axis(1)
)
, Format_String = "#,###"
